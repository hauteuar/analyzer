Sample Workflow Definition (YAML)
# Compliance Violation Check Workflow
# Created via No-Code Builder - Exported as YAML

id: "wf-compliance-check"
name: "Day Trade Violation Checker"
version: "1.0.0"
created_by: "Visual Builder"

trigger:
  type: "chatbot"
  intent: "check_violations"
  platforms: ["slack", "teams", "web"]
  entities:
    - name: "account_id"
      required: true
    - name: "date_range"
      default: "today"

nodes:
  - id: "start"
    type: "trigger"
    next: "get_account"

  - id: "get_account"
    type: "mcp_action"
    mcp: "oracle-trading"
    action: "executeQuery"
    params:
      sql: "SELECT * FROM accounts WHERE id = :account_id"
    next: ["check_day_trade", "check_90_day"]

  - id: "check_day_trade"
    type: "mcp_action"
    mcp: "db2-compliance"
    action: "callProcedure"
    params:
      procedure: "SP_CHECK_DAY_TRADE"
    next: "aggregate"

  - id: "check_90_day"
    type: "mcp_action"
    mcp: "db2-compliance"
    action: "callProcedure"
    params:
      procedure: "SP_CHECK_90_DAY_RULE"
    next: "aggregate"

  - id: "aggregate"
    type: "ai_node"
    action: "summarize"
    prompt: "Summarize compliance results with severity levels"
    next: "respond"

  - id: "respond"
    type: "response"
    format: "card"
    template: "compliance_summary"

API Specification
# OpenAPI 3.0 Specification

POST /api/v1/connector/trigger
# Trigger a workflow from your bot

Request Body:
{
  "source": "slack" | "teams" | "custom",
  "user_context": {
    "user_id": "U123456",
    "channel_id": "C789",
    "roles": ["tech_team", "admin"]
  },
  "payload": {
    "message": "Check day trade violations for account 12345"
  },
  "callback_url": "https://your-bot.com/callback",
  "async_mode": true
}

Response (202 Accepted):
{
  "execution_id": "exec-abc-123",
  "status": "processing"
}

Callback POST to your URL:
{
  "execution_id": "exec-abc-123",
  "status": "completed",
  "result": {
    "summary": "Found 3 day trade violations...",
    "data": { ... }
  }
}
Integration Code Example
// Integrate your existing bot with the Workflow Platform

const triggerWorkflow = async (userMessage, context) => {
  const response = await fetch('https://workflow-api.bofa.com/api/v1/connector/trigger', {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
      'X-API-Key': process.env.WORKFLOW_API_KEY
    },
    body: JSON.stringify({
      source: 'custom',
      user_context: {
        user_id: context.userId,
        channel_id: context.channelId,
        roles: context.roles
      },
      payload: {
        message: userMessage
      },
      callback_url: 'https://your-bot.bofa.com/webhook/result',
      async_mode: true
    })
  });
  
  return response.json();
};

// Handle async callback
app.post('/webhook/result', (req, res) => {
  const { execution_id, status, result } = req.body;
  
  // Send result back to user via your bot platform
  sendToUser(result.channel_id, {
    type: 'card',
    title: 'Workflow Result',
    body: result.summary,
    data: result.data
  });
  
  res.status(200).send('OK');
});
